def get_input(model, name):
    for elem in model.graph.input:
        if elem.name == name:
            return elem

def get_node(model, name):
    for elem in model.graph.node:
        if elem.name == name:
            return elem

def get_output(model, name):
    for elem in model.graph.output:
        if elem.name == name:
            return elem
